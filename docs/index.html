class Circle {
    constructor(radius) {
        this.radius = radius;
    }

    /**
     * Calculate the circumference of the circle.
     *
     * @return {number}
     */
    circumference() {
        return 2 * Math.PI * this.radius;
    }
}

const c = new Circle(0.5);
console.log(c.circumference()); // 3.1415...
